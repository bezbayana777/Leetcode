// 20 Valid parentheses
class Solution:
    def isValid(self, s: str) -> bool:
        stack = []
        closed_cnt = 0
        for elem in s:
            if elem in '([{':
                stack.append(elem)
            elif elem == ')' and stack and stack[-1]=='(':
                stack.pop()
            elif elem == ']' and stack and stack[-1] == '[':
                stack.pop()
            elif elem == '}' and stack and stack[-1] == '{':
                stack.pop()
            elif elem in ']})':
                closed_cnt+=1
            
        return True if (len(stack)+closed_cnt)==0 else False


// convertToBase7
class Solution:
    def convertToBase7(self, num: int) -> str:
        if num == 0: return "0"
        isPos = True if num > 0 else False
        base7Num = ''
        num = abs(num)
        while abs(num) > 0:
            base7Num = str(num%7) + base7Num
            num//=7
        return base7Num if isPos else '-' + base7Num

3110. Score of a String
class Solution:
    def scoreOfString(self, s: str) -> int:
        if len(s) == 1: return ord(s[0])
        sm = 0
        for i in range(1, len(s)):
            sm += abs(ord(s[i-1]) - ord(s[i]))
        return sm
        
2942. Find Words Containing Character
class Solution:
    def findWordsContaining(self, words: List[str], x: str) -> List[int]:
        indexes_list = []
        for ind, word in enumerate(words):
            if x in word:
                indexes_list.append(ind)
        return indexes_list
